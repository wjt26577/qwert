clip_changed(data_type) {
    try {
        global clipboard_history, max_history

        ; 只处理文本类型
        if (data_type != 1)
            return

        content := A_Clipboard
        
        ; 跳过空内容
        if (content = "")
            return

        ; 插入到开头（最新的在前）
        clipboard_history.InsertAt(1, content)

        ; 超出上限则删除最旧的
        while (clipboard_history.Length > max_history)
            clipboard_history.Pop()
    } catch
        throw
}

 

; --- 保存函数 ---
save_clipboard_history(*) {
    global clipboard_history  
    try {
        clip_map := Map()

        loop 10 {
            if clipboard_history.has(a_index)
                clip_map[a_index] := clipboard_history[a_index]
           else
                clip_map[a_index] := "nothing"
            
        }

        ; 直接覆盖保存
        json_text := json.dumps(clip_map, 2)
        json_file := FileOpen("settings\clipboard_history.json", "w") 
        json_file.Write(json_text)
        json_file.Close() 

    } catch as err {
        MsgBox(Format("Error: {1}`nFile: {2}`nLine:  {3}", err.message, err.file, err.Line))
    }    
}

clip_changed(data_type) {
    try {
        global clipboard_history, max_history

        ; 只处理文本类型
        if (data_type != 1)
            return

        content := A_Clipboard
        
        ; 跳过空内容
        if (content = "")
            return

        ; 插入到开头（最新的在前）
        clipboard_history.InsertAt(1, content)

        ; 超出上限则删除最旧的
        while (clipboard_history.Length > max_history)
            clipboard_history.Pop()
    } catch
        throw
}

 

; --- 保存函数 ---
save_clipboard_history(*) {
    global clipboard_history  
    try {
        clip_map := Map()

        loop 10 {
            if clipboard_history.has(a_index)
                clip_map[a_index] := clipboard_history[a_index]
           else
                clip_map[a_index] := "nothing"
            
        }

        ; 直接覆盖保存
        json_text := json.dumps(clip_map, 2)
        json_file := FileOpen("settings\clipboard_history.json", "w") 
        json_file.Write(json_text)
        json_file.Close() 

    } catch as err {
        MsgBox(Format("Error: {1}`nFile: {2}`nLine:  {3}", err.message, err.file, err.Line))
    }    
}

; DpiTest()
; MsgBox DllCall("GetDpiForWindow", "Ptr", DllCall("WindowFromPoint", "Int64", 0, "Ptr"))

DpiTest() {
    ; ============ 进程级 DPI 感知模式 ============
    awareness := _GetProcessDpiAwareness()
    awarenessName := Map(
        0, "UNAWARE (系统缩放)",
        1, "SYSTEM_AWARE (仅主屏)",
        2, "PER_MONITOR_AWARE (逐屏)"
    )
    
    ; ============ 主屏 DPI（GDI 方式） ============
    hdc := DllCall("GetDC", "ptr", 0, "ptr")
    gdiDpiX := DllCall("GetDeviceCaps", "ptr", hdc, "int", 88)  ; LOGPIXELSX
    gdiDpiY := DllCall("GetDeviceCaps", "ptr", hdc, "int", 90)  ; LOGPIXELSY
    DllCall("ReleaseDC", "ptr", 0, "ptr", hdc)
    
    ; ============ 系统 DPI（Win10+） ============
    sysDpi := 96
    try sysDpi := DllCall("GetDpiForSystem", "uint")
    
    ; ============ 汇总输出 ============
    out := "====== DPI 诊断报告 ======`n`n"
    out .= Format("进程 DPI 感知: {} ({})`n", awareness, awarenessName.Has(awareness) ? awarenessName[awareness] : "未知")
    out .= Format("系统 DPI (GetDpiForSystem): {}`n", sysDpi)
    out .= Format("主屏 GDI DPI: {}x{}`n", gdiDpiX, gdiDpiY)
    out .= Format("A_ScreenDPI: {}`n", A_ScreenDPI)
    out .= Format("主显示器序号: {}`n`n", MonitorGetPrimary())
    
    ; ============ 逐显示器信息 ============
    count := MonitorGetCount()
    out .= Format("显示器数量: {}`n", count)
    out .= "─────────────────────────────`n"
    
    loop count {
        mon := A_Index
        MonitorGet(mon, &l, &t, &r, &b)
        MonitorGetWorkArea(mon, &wl, &wt, &wr, &wb)
        
        isPrimary := (mon = MonitorGetPrimary()) ? " ★主屏" : ""
        name := MonitorGetName(mon)
        
        w := r - l
        h := b - t
        
        ; 获取 HMONITOR
        hMon := _MonitorHandleFromRect(l, t, r, b)
        
        ; GetDpiForMonitor
        monDpiX := 0, monDpiY := 0
        effDpiOk := _GetDpiForMonitor(hMon, 0, &monDpiX, &monDpiY)   ; MDT_EFFECTIVE_DPI
        
        angDpiX := 0, angDpiY := 0
        angDpiOk := _GetDpiForMonitor(hMon, 1, &angDpiX, &angDpiY)   ; MDT_ANGULAR_DPI
        
        rawDpiX := 0, rawDpiY := 0
        rawDpiOk := _GetDpiForMonitor(hMon, 2, &rawDpiX, &rawDpiY)   ; MDT_RAW_DPI
        
        ; 缩放百分比
        scalePct := monDpiX > 0 ? Round(monDpiX / 96 * 100) : "?"
        
        ; 相对主屏的补偿系数
        compensate := monDpiX > 0 && gdiDpiX > 0 ? Round(monDpiX / gdiDpiX, 3) : "?"
        
        out .= Format("`n【显示器 #{}】{}{}`n", mon, isPrimary, name ? " (" name ")" : "")
        out .= Format("  分辨率: {}x{}`n", w, h)
        out .= Format("  坐标: L={} T={} R={} B={}`n", l, t, r, b)
        out .= Format("  工作区: L={} T={} R={} B={}`n", wl, wt, wr, wb)
        out .= Format("  Effective DPI: {}x{} ({}%缩放) {}`n", monDpiX, monDpiY, scalePct, effDpiOk ? "" : "⚠失败")
        out .= Format("  Angular DPI:   {}x{} {}`n", angDpiX, angDpiY, angDpiOk ? "" : "⚠失败")
        out .= Format("  Raw DPI:       {}x{} {}`n", rawDpiX, rawDpiY, rawDpiOk ? "" : "⚠失败")
        out .= Format("  HMONITOR: {}`n", hMon)
        out .= Format("  补偿系数 (monDpi/primaryDpi): {}`n", compensate)
        out .= Format("  → 若主屏字号30, 此屏应设: {}`n", monDpiX > 0 && gdiDpiX > 0 ? Round(30 * monDpiX / gdiDpiX) : "?")
    }
    
    ; ============ 鼠标所在屏 ============
    out .= "`n─────────────────────────────`n"
    CoordMode("Mouse", "Screen")
    MouseGetPos(&mx, &my)
    curMon := _MonitorFromPoint(mx, my)
    out .= Format("鼠标位置: ({}, {}) → 显示器 #{}`n", mx, my, curMon)
    
    ; ============ 显示结果 ============
    g := Gui("+AlwaysOnTop -DPIScale", "DPI 诊断")
    g.SetFont("s11", "Consolas")
    g.Add("Edit", "w650 h550 ReadOnly -Wrap", out)
    
    btnCopy := g.Add("Button", "w120", "复制到剪贴板")
    btnCopy.OnEvent("Click", (*) => (A_Clipboard := out, ToolTip("已复制"), SetTimer((*) => ToolTip(), -1500)))
    
    btnRefresh := g.Add("Button", "x+10 w120", "刷新")
    btnRefresh.OnEvent("Click", (*) => (g.Destroy(), DpiTest()))
    
    g.Show()
    return
    
    
    ; ============ 内部函数 ============
    
    _GetProcessDpiAwareness() {
        awareness := 0
        try DllCall("Shcore\GetProcessDpiAwareness", "ptr", 0, "int*", &awareness)
        return awareness
    }
    
    _MonitorHandleFromRect(l, t, r, b) {
        ; 取中心点 → MonitorFromPoint
        cx := (l + r) // 2
        cy := (t + b) // 2
        pt64 := cx | (cy << 32)
        return DllCall("MonitorFromPoint", "int64", pt64, "uint", 2, "ptr")  ; MONITOR_DEFAULTTONEAREST
    }
    
    _GetDpiForMonitor(hMon, type, &dpiX, &dpiY) {
        dpiX := 0, dpiY := 0
        try {
            hr := DllCall("Shcore\GetDpiForMonitor", "ptr", hMon, "uint", type,
                          "uint*", &dpiX, "uint*", &dpiY)
            return hr = 0
        }
        return false
    }
    
    _MonitorFromPoint(x, y) {
        loop MonitorGetCount() {
            MonitorGet(A_Index, &l, &t, &r, &b)
            if (x >= l && x < r && y >= t && y < b)
                return A_Index
        }
        return MonitorGetPrimary()
    }
}
